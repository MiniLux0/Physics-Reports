\section{Propagación de incertidumbre}

% Reemplazar con el cálculo real del lab.
% Fórmula base, derivadas parciales, sustitución numérica, resultado final.

Para $a_G = 2d/t^2$, aplicando suma en cuadratura:

\begin{equation}
    \delta a_G = \sqrt{
        \left(\frac{2}{t^2}\,\delta d\right)^2 +
        \left(\frac{4d}{t^3}\,\delta t\right)^2
    }
\end{equation}

Resultado:

\[
    \boxed{a_G = (3{,}50 \pm 0{,}07) \times 10^{-3}\ \unit{m/s^2}}
\]
