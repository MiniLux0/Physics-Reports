\section{Propagación de incertidumbre}
\label{anx:prop_incertidumbre}
%% ============================================================

Se aplica la regla de suma en cuadratura en todos los casos. Las
incertidumbres instrumentales empleadas son:
$\delta R = \qty{0.005}{\centi\meter}$ (vernier),
$\delta m = \qty{0.1}{\gram}$ (balanza de precisión),
$\delta n = 1$ división (brazo mayor).
Se adopta $g = \qty{9.81}{\meter\per\second\squared}$.

%% ------------------------------------------------------------------
\subsection{Longitud de contacto \texorpdfstring{$L$}{L}}

%% ------------------------------------------------------------------

Para $L = 2\pi(R_{\text{ext}} + R_{\text{int}})$, con $R_{\text{ext}}$
y $R_{\text{int}}$ medidos con el mismo instrumento:

\begin{equation}
    \delta L = 2\pi\sqrt{(\delta R_{\text{ext}})^2 + (\delta R_{\text{int}})^2}
             = 2\pi\sqrt{2}\,\delta R
\end{equation}

Sustituyendo $\delta R = \qty{5e-5}{\meter}$:

\begin{equation*}
    \delta L = 2\pi\sqrt{2} \times \num{5e-5}
             = \qty{4.44e-4}{\meter}
\end{equation*}

\begin{equation*}
    \boxed{L = \qty{6.2957 \pm 0.0004e-1}{\meter}}
\end{equation*}

%% ------------------------------------------------------------------
\subsection{Masa efectiva \texorpdfstring{$M_{\text{ef}}$}{Mef}}
%% ------------------------------------------------------------------

Para $M_{\text{ef}} = \sum_j m_j\,n_j/10$, propagando la incertidumbre
en cada masa $m_j$ (la incertidumbre en $n_j$ se considera despreciable
frente a $\delta m$):

\begin{equation}
    \delta M_{\text{ef}} = \frac{\delta m}{10}
    \sqrt{\sum_j n_j^2}
\end{equation}

Cálculo representativo — Medición~1
($J_1$ en $n=1$, $J_4$ en $n=3$):

\begin{equation*}
    \delta M_{\text{ef},1}
    = \frac{\num{0.1e-3}}{10}\sqrt{1^2 + 3^2}
    = \num{1e-5}\sqrt{10}
    = \qty{3.16e-5}{\kilogram}
\end{equation*}

%% ------------------------------------------------------------------
\subsection{Coeficiente de tensión superficial \texorpdfstring{$\gamma_i$}{gamma\_i}}
%% ------------------------------------------------------------------

Para $\gamma_i = M_{\text{ef},i}\,g\,/\,L$, aplicando suma en
cuadratura:

\begin{equation}
    \frac{\delta\gamma_i}{\gamma_i} =
    \sqrt{
        \left(\frac{\delta M_{\text{ef},i}}{M_{\text{ef},i}}\right)^2 +
        \left(\frac{\delta L}{L}\right)^2
    }
\end{equation}

Cálculo representativo — Medición~1
($M_{\text{ef},1} = \qty{2.38e-3}{\kilogram}$,
$\delta M_{\text{ef},1} = \qty{3.16e-5}{\kilogram}$):

\begin{equation*}
    \frac{\delta\gamma_1}{\gamma_1}
    = \sqrt{
        \left(\frac{\num{3.16e-5}}{\num{2.38e-3}}\right)^2 +
        \left(\frac{\num{4.44e-4}}{\num{6.2957e-1}}\right)^2
      }
    = \sqrt{(\num{0.01328})^2 + (\num{0.000705})^2}
    = \num{0.01330}
\end{equation*}

\begin{equation*}
    \delta\gamma_1 = \num{0.01330} \times \num{37.09} = \qty{0.49}{\milli\newton\per\meter}
\end{equation*}

%% ------------------------------------------------------------------
\subsection{Incertidumbre del promedio \texorpdfstring{$\overline{\gamma}_1$}{gamma promedio}}
%% ------------------------------------------------------------------

La incertidumbre del promedio se estima mediante la desviación estándar
de la media:

\begin{equation}
    \delta\overline{\gamma}_1
    = \sqrt{\frac{\displaystyle\sum_{i=1}^{N}(\gamma_i - \overline{\gamma}_1)^2}
                 {N(N-1)}}
\end{equation}

Sustituyendo con $N = 5$ y
$\overline{\gamma}_1 = \qty{39.73}{\milli\newton\per\meter}$:

\begin{equation*}
    \sum_{i=1}^{5}(\gamma_i - \overline{\gamma}_1)^2
    = (\num{37.09}-\num{39.73})^2 + (\num{41.60}-\num{39.73})^2 + (\num{38.33}-\num{39.73})^2
      + (\num{40.20}-\num{39.73})^2 + (\num{41.45}-\num{39.73})^2
\end{equation*}

\begin{equation*}
    = \num{6.9696} + \num{3.4969} + \num{1.9600} + \num{0.2209} + \num{2.9584}
    = \qty{15.608}{\milli\newton\squared\per\meter\squared}
\end{equation*}

\begin{equation*}
    \delta\overline{\gamma}_1
    = \sqrt{\frac{\num{15.608}}{5 \times 4}}
    = \sqrt{\num{0.7804}}
    = \qty{0.88}{\milli\newton\per\meter}
\end{equation*}

\begin{equation*}
    \boxed{\overline{\gamma}_1 = \qty{39.73 \pm 0.88}{\milli\newton\per\meter}}
\end{equation*}

%% ------------------------------------------------------------------
\subsection{Coeficiente de tensión superficial \texorpdfstring{$\gamma_2$}{gamma\_2}}
%% ------------------------------------------------------------------

Para $\gamma_2 = P\,/\,[2(h^2/(a-b)+a+b)]$, las variables
independientes con incertidumbre son $m$, $a$, $b$ y $h$.

Primero se calcula el peso y su incertidumbre:

\begin{equation*}
    P = mg = \num{1.2e-3} \times \num{9.81} = \qty{11.772e-3}{\newton}
\end{equation*}

\begin{equation*}
    \delta P = \delta m \cdot g
    = \num{0.1e-3} \times \num{9.81}
    = \qty{0.981e-3}{\newton}
\end{equation*}

Definiendo $Q = h^2/(a-b) + a + b$, las derivadas parciales son:

\begin{equation}
    \frac{\partial Q}{\partial a} = -\frac{h^2}{(a-b)^2} + 1, \qquad
    \frac{\partial Q}{\partial b} =  \frac{h^2}{(a-b)^2} + 1, \qquad
    \frac{\partial Q}{\partial h} = \frac{2h}{a-b}
\end{equation}

Sustituyendo $a = \qty{2.73}{\centi\meter}$,
$b = \qty{2.55}{\centi\meter}$,
$h = \qty{2.78}{\centi\meter}$,
$a - b = \qty{0.18}{\centi\meter}$,
$\delta a = \delta b = \delta h = \qty{0.005}{\centi\meter}$:

\begin{equation*}
    \frac{\partial Q}{\partial a}
    = -\frac{(\num{2.78})^2}{(\num{0.18})^2} + 1
    = -\frac{\num{7.7284}}{\num{0.0324}} + 1
    = -\num{238.53} + 1
    = -\num{237.53}
\end{equation*}

\begin{equation*}
    \frac{\partial Q}{\partial b}
    = \frac{\num{7.7284}}{\num{0.0324}} + 1
    = \num{238.53} + 1
    = \num{239.53}
\end{equation*}

\begin{equation*}
    \frac{\partial Q}{\partial h}
    = \frac{2 \times \num{2.78}}{\num{0.18}}
    = \num{30.89}
\end{equation*}

\begin{equation*}
    \delta Q = \sqrt{
        (\num{237.53} \times \num{0.005})^2 +
        (\num{239.53} \times \num{0.005})^2 +
        (\num{30.89}  \times \num{0.005})^2
    }\ \si{\centi\meter}
\end{equation*}

\begin{equation*}
    \delta Q = \sqrt{\num{1.4077} + \num{1.4328} + \num{0.0238}}
             = \sqrt{\num{2.8643}}
             = \qty{1.693}{\centi\meter}
             = \qty{1.693e-2}{\meter}
\end{equation*}

Finalmente, para $\gamma_2 = P/(2Q)$:

\begin{equation}
    \frac{\delta\gamma_2}{\gamma_2}
    = \sqrt{
        \left(\frac{\delta P}{P}\right)^2 +
        \left(\frac{\delta Q}{Q}\right)^2
      }
\end{equation}

\begin{equation*}
    \frac{\delta\gamma_2}{\gamma_2}
    = \sqrt{
        \left(\frac{\num{0.981e-3}}{\num{11.772e-3}}\right)^2 +
        \left(\frac{\num{1.693e-2}}{\num{4.822e-1}}\right)^2
      }
    = \sqrt{(\num{0.0833})^2 + (\num{0.0351})^2}
    = \num{0.0904}
\end{equation*}

\begin{equation*}
    \delta\gamma_2 = \num{0.0904} \times \num{12.21}
                  = \qty{1.10}{\milli\newton\per\meter}
\end{equation*}

\begin{equation*}
    \boxed{\gamma_2 = \qty{12.21 \pm 1.10}{\milli\newton\per\meter}}
\end{equation*}
