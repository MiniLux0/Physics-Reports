\section{Materiales y Equipos}

Para la realización del presente experimento se emplearon los siguientes materiales y equipos:

\subsection{Aparato de dilatación térmica lineal (Dilatómetro)}
\begin{itemize}
    \item \textbf{Descripción y función:} Soporte metálico rígido diseñado para sostener el tubo de ensayo en posición horizontal. Cuenta con un punto de sujeción fijo por medio de una pinza de ajuste en un extremo, mientras que el extremo libre descansa sobre una aguja cilíndrica de rodamiento. Esto permite que la expansión lineal del tubo se traduzca de forma directa en la rotación de la aguja.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico
        \item Capacidad: muestras de hasta \qty{800}{\milli\meter} de longitud
    \end{itemize}
    \item \textbf{Precauciones:} Asegurar que la alineación del tubo con la aguja de rodamiento sea completamente perpendicular, garantizando condiciones de rodadura pura sin deslizamientos.
\end{itemize}

    \caption{Aparato de dilatación térmica lineal (dilatómetro) utilizado para sostener las muestras.}

\subsection{Generador de vapor de agua}
\begin{itemize}
    \item \textbf{Descripción y función:} Recipiente calefactor de agua (matraz de ebullición) provisto de una resistencia eléctrica o calentador y una manguera flexible de caucho. Su función es producir e inyectar vapor continuo a través del interior de los tubos para calentarlos de manera homogénea hasta la temperatura de ebullición.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico
        \item Temperatura nominal de operación: 100$^\circ$C
    \end{itemize}
    \item \textbf{Precauciones:} Manipular las conexiones con pinzas térmicas para evitar quemaduras por vapor caliente y verificar la estanqueidad de las mangueras de acople.
\end{itemize}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.50\linewidth, height=5.5cm]{fig_generador_vapor}
    \caption{Generador de vapor utilizado como fuente térmica para el calentamiento del tubo.}
    \label{fig:generador_vapor}
\end{figure}

\subsection{Calibrador Vernier}
\begin{itemize}
    \item \textbf{Descripción y función:} Instrumento de medición dimensional utilizado para determinar con alta precisión el diámetro $D$ de la aguja cilíndrica de rodamiento.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico
        \item Incertidumbre instrumental: $\delta D = \qty{\pm 0.025}{\milli\meter}$ (lo que define una incertidumbre en el radio de la aguja de $\delta R = \qty{\pm 0.0125}{\milli\meter}$)
    \end{itemize}
    \item \textbf{Precauciones:} Limpiar las mordazas de medición y verificar que el instrumento marque cero antes de realizar las lecturas para evitar errores de cero.
\end{itemize}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.50\linewidth, height=5.5cm]{fig_vernier}
    \caption{Calibrador vernier empleado para la medición del diámetro de la aguja.}
    \label{fig:vernier}
\end{figure}

\subsection{Regla metálica graduada}
\begin{itemize}
    \item \textbf{Descripción y función:} Regla graduada milimétrica de un metro de longitud, utilizada para medir la longitud inicial libre del tubo ($L_0$) entre sus apoyos de sujeción.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico
        \item Incertidumbre instrumental: $\delta L_0 = \qty{\pm 0.05}{\milli\meter}$
    \end{itemize}
    \item \textbf{Precauciones:} Alinear perpendicularmente la visual con la escala graduada al tomar la lectura para minimizar los errores sistemáticos causados por paralaje.
\end{itemize}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.50\linewidth, height=5.5cm]{fig_regla}
    \caption{Regla graduada metálica empleada para medir la longitud inicial libre de las muestras.}
    \label{fig:regla}
\end{figure}

\subsection{Transportador de ángulos}
\begin{itemize}
    \item \textbf{Descripción y función:} Dial graduado angular acoplado al eje de la aguja indicadora de rodamiento. Su función es registrar el ángulo de giro angular $\Delta\theta$ que experimenta la aguja indicadora durante el proceso de expansión térmica del tubo.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico
        \item Incertidumbre instrumental: $\delta\theta = \qty{\pm 0.5}{\degree}$ (equivalente a $\delta\theta = \qty{\pm 0.00873}{\radian}$)
    \end{itemize}
    \item \textbf{Precauciones:} Asegurarse de que el indicador de aguja esté perfectamente alineado en posición vertical hacia abajo (cero de referencia angular) al comenzar cada ensayo.
\end{itemize}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.50\linewidth, height=5.5cm]{fig_transportador}
    \caption{Transportador de ángulos acoplado al eje del dial indicador.}
    \label{fig:transportador}
\end{figure}

\subsection{Termómetro}
\begin{itemize}
    \item \textbf{Descripción y función:} Termómetro de precisión empleado para medir la temperatura ambiente del laboratorio.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Tipo: analógico de columna líquida
        \item Incertidumbre: $\delta T = \pm 0{,}5^\circ$C
    \end{itemize}
    \item \textbf{Precauciones:} Evitar que el termómetro entre en contacto directo con corrientes de aire externas al medir la temperatura inicial del laboratorio.
\end{itemize}

    \centering

    \caption{Termómetro de precisión empleado para medir la temperatura ambiente.}

\subsection{Tubos de ensayo (muestras)}
\begin{itemize}
    \item \textbf{Descripción y función:} Tubos cilíndricos huecos de aluminio, cobre y vidrio borosilicato que conducen el vapor caliente a lo largo de su interior para experimentar y medir su dilatación lineal térmica.
    \item \textbf{Especificaciones:}
    \begin{itemize}
        \item Materiales: aluminio comercial, cobre y vidrio borosilicato
        \item Longitudes nominales: en el rango de \qtyrange{700}{770}{\milli\meter}
    \end{itemize}
    \item \textbf{Precauciones:} Manipular las muestras metálicas y de vidrio con guantes aislantes térmicos durante y después del flujo de vapor para evitar quemaduras por contacto.
\end{itemize}
